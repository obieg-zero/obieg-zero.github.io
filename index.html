import { createFlow, templateNode, extractNode } from '@obieg-zero/core'
import { ocrNode } from '@obieg-zero/ocr'
import { llmNode } from '@obieg-zero/llm'

const flow = createFlow()

flow.node('ocr', ocrNode({ language: 'pol' }))
flow.node('prompt', templateNode({
  template: `Extract from document:\n{{context}}\n\nReturn JSON: {amount, date, party}`
}))
flow.node('llm', llmNode({ modelUrl: '/models/Bielik-1.5B.gguf' }))
flow.node('parse', extractNode({ output: 'result' }))

flow.set('file', uploadedPdf)
await flow.run('ocr', 'prompt', 'llm', 'parse')

flow.get('result')
// → { amount: 350000, date: "2019-03-15", party: "mBank S.A." }
