Use case — PDF to answer, fully local

  • Upload PDF
    Stored in OPFS — no server, no upload
  • OCR
    pdfjs text layer → Tesseract fallback → pages[]
  • Template
    Inject extracted text into prompt via {{context}}
  • LLM
    Bielik 1.5B runs in browser via WebAssembly
  • Answer
    flow.get('answer') — structured or free-text
import { createFlow, templateNode } from '@obieg-zero/core'
import { opfsUpload } from '@obieg-zero/storage'
import { ocrNode } from '@obieg-zero/ocr'
import { llmNode } from '@obieg-zero/llm'

const flow = createFlow()

flow.node('upload', opfsUpload())
flow.node('ocr',    ocrNode({ language: 'pol' }))
flow.node('prompt', templateNode({
  template: 'Podsumuj dokument:\n{{context}}'
}))
flow.node('llm', llmNode({
  modelUrl: '/models/Bielik-1.5B.gguf'
}))

// run the pipeline
flow.set('file', userFile)
flow.set('projectId', 'demo')
flow.set('fileKey', 'doc.pdf')

await flow.run(
  'upload', 'ocr', 'prompt', 'llm'
)

console.log(flow.get('answer'))
