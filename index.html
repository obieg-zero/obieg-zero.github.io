  return <div>Mój moduł</div>
}

registerModule({
  id: 'my-module',

  icon: MyIcon,

Flow — pipeline w 50 LOC

import { createFlow, templateNode, extractNode } from '@obieg-zero/core'
import { opfsUpload, persistSave, persistLoad } from '@obieg-zero/storage'
import { ocrNode } from '@obieg-zero/ocr'
import { embedNode, searchNode } from '@obieg-zero/embed'
import { llmNode } from '@obieg-zero/llm'

const flow = createFlow()

flow.node('upload', opfsUpload())
flow.node('ocr',    ocrNode({ language: 'pol' }))
flow.node('embed',  embedNode({ model: 'Xenova/multilingual-e5-small', ... }))
flow.node('search', searchNode({ topK: 3 }))
flow.node('llm',    llmNode({ modelUrl: '/models/Bielik-1.5B.gguf' }))
flow.node('save',   persistSave({ keys: ['pages', 'chunks'] }))

// run pipeline
flow.set('file', pdf)
await flow.run('upload', 'ocr', 'embed', 'save')
// → flow.get('pages'), flow.get('chunks')
